#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> num_list, int n) {
    vector<int> answer;

int i = n;
while (i < num_list.size()) {
    answer.push_back(num_list[i]);
    i++;
}

i = 0;
while (i < n) {
    answer.push_back(num_list[i]);
    i++;
}

return answer;
}
