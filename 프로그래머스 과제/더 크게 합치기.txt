#include <string>
#include <vector>

using namespace std;

int solution(int a, int b) {
    int answer = 0;

    int i = 1;
    while (i <= b) {
        i *= 10;
    }

    answer = a * i + b;

    i = 1;
    while (i <= a) {
        i *= 10;
    }

    int ba = b * i + a;

    if (answer < ba) {
        answer = ba;
    }

    return answer;
}
